以下是具体的实现，通过requests get 到就业信息网的html结构，然后查找所有 div.arcList下li元素的文本内容 写入 info.txt:

# -*-coding:utf-8 -*-
from bs4 import BeautifulSoup 
import requests
import codecs

def get_job_info():
    url='http://job.xidian.edu.cn/html/zpxx/jobs/list_3_2.html'
    res=requests.get(url)
    info=res.content
    soup=BeautifulSoup(info,'html.parser')
    content=codecs.open('info.txt','w','utf-16')
    for div in soup.find_all('div',{'class':'arcList'}):
        for li in div.find_all('li'):
            cc=li.text
            content.write(cc+'\n')
    content.close()
if __name__=='__main__':
    get_job_info()
