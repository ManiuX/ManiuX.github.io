写一个简单的自动化测试例子如下图 测试套件中包括一个百度搜索的测试用例和一个在输入框上传文件的测试用例：

  • BaiduSearch

实现的功能非常简单 百度搜索关键字 PG one 下面是测试用例BaiduSearch.py的具体实现：


# -*- coding: utf-8 -*-
from selenium import webdriver
import unittest,time

class Bai(unittest.TestCase):
    '''baidu test'''
    def setUp(self):
        self.driver=webdriver.Chrome()
        self.driver.maximize_window()
        self.driver.implicitly_wait(20)
        print('baidu test start')
    
    def test_bai(self):
        '''keyword PG one'''
        driver=self.driver
        driver.get('http://www.baidu.com')
        driver.find_element_by_id('kw').send_keys('PG one')
        driver.find_element_by_id('su').click()
        title=driver.title
        driver.get_screenshot_as_file('D:\\baidu.jpg')
        title=driver.title
        try:
            self.assertEqual(title,u'PG one_百度搜索')
            print ('The page title:'+title)
        except:
            print(title)
            print('Get the wrong title')
    def tearDown(self):
        self.driver.quit()
        print('baidu test end')


if __name__=='__main__':
    unittest.main()
  • UploadFile 也是一个非常简单的功能，在菜鸟教程官网提供的html教学页面上实现一个input 输入框的文件上传功能UploadFile.py 具体实现如下：
# -*- coding: utf-8 -*-
from selenium import webdriver
import time,unittest

class Cai(unittest.TestCase):
    '''cainiao test'''
    def setUp(self):
        self.driver=webdriver.Chrome()
        self.driver.maximize_window()
        self.driver.implicitly_wait(10)
    
    def test_cai(self):
        '''upload file input'''
        self.driver.get('http://www.runoob.com/try/try.php?filename=tryhtml5_input_type_file')
        self.driver.switch_to_frame('iframeResult')
        self.driver.find_element_by_name('img').send_keys('D:\sj\demo.js')
        time.sleep(3)
        js='window.scrollTo(0,500)'
        self.driver.execute_script(js)
        self.driver.get_screenshot_as_file('D:\\cai.jpg')
    
    def tearDown(self):
        self.driver.quit()

if __name__=='__main__':
    unittest.main()
  • 测试套件集成 接下来就是将以上两个测试用例集成在测试套件中，并设置生成测试报告runtest.py 具体实现如下：
# -*- coding: utf-8 -*-
import unittest,sys
sys.path.append('./testcase')
import BaiduSearch,UploadFile
from HTMLTestRunner import HTMLTestRunner

suite=unittest.TestSuite()
suite.addTest(BaiduSearch.Bai('test_bai'))
suite.addTest(UploadFile.Cai('test_cai'))

if __name__=='__main__':
    now=time.strftime("%Y-%m-%d %H-%M-%S")
    fp=open('./re{e}.html'.format(e=now),'wb')
    runner=HTMLTestRunner(stream=fp,title='first case',description='the test runner result')
    runner.run(suite)
    fp.close()
