双色球规则

1-33红球抽六个，1-16蓝球抽一个：

嗯，买张彩票 没中！ 写了一个简单的页面放进去啦 双色球

js代码如下：

function getNum(){
	var red=[],blue=[],res=[];
	for (let i=1;i<34;i++){
		red.push(i);
	}
	for (let i=1;i<17;i++){
		blue.push(i);
	}
	for(let i=0;i<6;i++){
		n=32-i;
		k=Math.ceil(Math.random()*n+0);
		res.push(red[k]);
		red.splice(k,1)
	}
	res.sort((a,b)=>(a-b));
	k2=Math.ceil(Math.random()*15+0);
	res.push(blue[k2]);
	return res
}
console.log(getNum());
